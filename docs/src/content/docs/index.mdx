import { Card, CardGrid, Code } from '@astrojs/starlight/components';

```ts
import { Brex } from 'brex';

const brex = new Brex({ token: process.env.BREX_TOKEN! });

// Typed responses, auto-pagination, idempotency — all built in.
for await (const expense of brex.expenses.list({ 'expand[]': ['merchant'] })) {
  console.log(expense.id, expense.merchant?.raw_descriptor);
}
```

  <Card title="Fully typed" icon="seti:typescript">
    Request bodies, query params, and responses are typed straight from the
    specs. Every method also carries the operation's full description inline.
  </Card>
  <Card title="Tree-shakeable" icon="puzzle">
    Import one API via a subpath (`brex/payments`) and the other nine never
    enter your bundle — about 3.5&nbsp;kB min for a single API.
  </Card>
  <Card title="Runs everywhere" icon="rocket">
    Native `fetch`, zero runtime dependencies, ESM-only. Node&nbsp;≥&nbsp;20.19
    (including `require()`), Bun, Deno, browsers, and edge runtimes.
  </Card>
  <Card title="Auto-pagination" icon="list-format">
    Every list method is both a promise and an async iterator — await a page,
    or `for await` across all of them.
  </Card>
  <Card title="Idempotency built in" icon="approve-check">
    `Idempotency-Key` headers are sent automatically wherever Brex requires
    them, with an auto-generated UUID by default.
  </Card>
  <Card title="LLM-friendly" icon="comment-alt">

  </Card>
